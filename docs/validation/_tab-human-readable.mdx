        Features:
        - Activity history across 35+ chains in real-time
        - Filterable by Receieved or Submitted
      </div>

      <div className="basis-2/5">

        query AccountTimeline($address: String!, $startTime: Int!, $endTime: Int!) {
          accountActivity(

            startTime: $startTime
            endTime: $endTime
          ) {
            transactions {
              timestamp
              type
              value
              description
            }
          }
        }

              value
